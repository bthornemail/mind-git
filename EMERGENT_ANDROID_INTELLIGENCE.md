```bash
# Install Emergent Intelligence

# Initialize on first phone
emergent init
```

```bash
# Custom node configuration

# Network optimization

# AI tuning
emergent config --ai-mode aggressive --energy-threshold 20

# Interface selection
emergent config --ui terminal,web,touch
```
